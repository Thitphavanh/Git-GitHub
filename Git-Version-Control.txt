# Local Git Repository

git init

git add xxx.html

git add *.html

git add .

git commit -m "first commit" 

git branch -M main

git remote add origin https://github.com/Thitphavanh/xxxx.git

git push -u origin main

# Git Repository (Remove)

git rm -r --cached .

git rm --cached xxx.html 

# Status Git Repository

git status

# Tracked Status

Modified

Staged

Committed
